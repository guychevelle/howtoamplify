type Process @model @auth(rules: [{allow: owner}, {allow: private: [read]}]) {
  id: ID!
  name: String!
  description: String
  picturekey: String
  steps: [Steps] @hasMany
}

type Steps @model @auth(rules: [{allow: owner}, {allow: private: [read]}]) {
  id: ID!
  stepnum: Int!
  name: String!
  description: String
  picturekey: String
  steptextkey: String
  code: Code @hasOne
}

type Code @model @auth(rules: [{allow: owner}, {allow: private: [read]}]) {
  id: ID!
  codetextkey: String
}
